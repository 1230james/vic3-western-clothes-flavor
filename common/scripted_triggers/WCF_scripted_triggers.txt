WCF_country_is_westernized = {
    OR = {
        has_variable = snori_clothes_decision_taken
        any_primary_culture = {
            has_discrimination_trait = european_heritage
        }
    }
}

WCF_can_start_westernizing = {
    OR = {
        # Option A: Progressives have sufficient clout
        AND = {
            # Traditionalists cannot be powerful
            AND = {
                ig:ig_landowners = {
                    is_powerful = no
                }
                ig:ig_devout = {
                    is_powerful = no
                }
                ig:ig_rural_folk = {
                    is_powerful = no
                }
                ig:ig_petty_bourgeoisie = {
                    is_powerful = no
                }
            }
            
            # Most progressives must be influential
            calc_true_if = {
                amount >= 3
                ig:ig_intelligentsia = {
                    is_marginal = no
                }
                ig:ig_industrialists = {
                    is_marginal = no
                }
                ig:ig_armed_forces = {
                    is_marginal = no
                }
                ig:ig_trade_unions = {
                    is_marginal = no
                }
            }
            
            # Progressives must have at least 1 powerful IG
            OR = {
                ig:ig_intelligentsia = {
                    is_powerful = yes
                }
                ig:ig_industrialists = {
                    is_powerful = yes
                }
                ig:ig_armed_forces = {
                    is_powerful = yes
                }
                ig:ig_trade_unions = {
                    is_powerful = yes
                }
            }
        }
        
        # Option B: Be a subject of a westernized nation
        AND = {
            is_subject = yes
            trigger_if = {
                limit = {
                    is_subject = yes
                }
                custom_tooltip = {
                    text = WCF_tt_overlord_westernized_sub
                    overlord = {
                        WCF_country_is_westernized = yes
                    }
                }
            }
            trigger_else = {
                custom_tooltip = {
                    text = WCF_tt_overlord_westernized_ind
                    overlord = {
                        WCF_country_is_westernized = yes
                    }
                }
            }
        }
    }
}
